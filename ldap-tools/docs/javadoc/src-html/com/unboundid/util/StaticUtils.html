254  /**
255   * Retrieves the set of currently defined system properties.  If possible,
256   * this will simply return the result of a call to
257   * {@code System.getProperties}.  However, the LDAP SDK is known to be used in
258   * environments where a security manager prevents setting system properties,
259   * and in that case, calls to {@code System.getProperties} will be rejected
260   * with a {@code SecurityException} because the returned structure is mutable
261   * and could be used to alter system property values.  In such cases, a new
262   * empty {@code Properties} object will be created, and may optionally be
263   * populated with the values of a specific set of named properties.
264   *
265   * @param  propertyNames  An optional set of property names whose values (if
266   *                        defined) should be included in the
267   *                        {@code Properties} object that will be returned if a
268   *                        security manager prevents retrieving the full set of
269   *                        system properties.  This may be {@code null} or
270   *                        empty if no specific properties should be retrieved.
271   *
272   * @return  The value returned by a call to {@code System.getProperties} if
273   *          possible, or a newly-created properties map (possibly including
274   *          the values of a specified set of system properties) if it is not
275   *          possible to get a mutable set of the system properties.
276   */
277  @NotNull()
278  public static Properties getSystemProperties(
279                                @Nullable final String... propertyNames)
280  {
281    try
282    {
283      final Properties properties = System.getProperties();
284
285      final String forceThrowPropertyName =
286           StaticUtils.class.getName() + ".forceGetSystemPropertiesToThrow";
287
288      // To ensure that we can get coverage for the code below in which there is
289      // a restrictive security manager in place, look for a system property
290      // that will cause us to throw an exception.
291      final Object forceThrowPropertyValue =
292           properties.getProperty(forceThrowPropertyName);
293      if (forceThrowPropertyValue != null)
294      {
295        throw new SecurityException(forceThrowPropertyName + '=' +
296             forceThrowPropertyValue);
297      }
298
299      return properties;
300    }
301    catch (final SecurityException e)
302    {
303      Debug.debugException(e);
304    }
305
306
307    // If we have gotten here, then we can assume that a security manager
308    // prevents us from accessing all system properties.  Create a new proper
309    final Properties properties = new Properties();
310    if (propertyNames != null)
311    {
312      for (final String propertyName : propertyNames)
313      {
314        final Object propertyValue = System.getProperty(propertyName);
315        if (propertyValue != null)
316        {
317          properties.put(propertyName, propertyValue);
318        }
319      }
320    }
321
322    return properties;
323  }

596  /**
597   * Retrieves a UTF-8 byte representation of the provided string.
598   *
599   * @param  s  The string for which to retrieve the UTF-8 byte representation.
600   *
601   * @return  The UTF-8 byte representation for the provided string.
602   */
603  @NotNull()
604  public static byte[] getBytes(@Nullable final String s)
605  {
606    final int length;
607    if ((s == null) || ((length = s.length()) == 0))
608    {
609      return NO_BYTES;
610    }
611
612    final byte[] b = new byte[length];
613    for (int i=0; i < length; i++)
614    {
615      final char c = s.charAt(i);
616      if (c <= 0x7F)
617      {
618        b[i] = (byte) (c & 0x7F);
619      }
620      else
621      {
622        return s.getBytes(StandardCharsets.UTF_8);
623      }
624    }
625
626    return b;
627  }

678   * Indicates whether the provided character is a printable ASCII character, as
679   * per RFC 4517 section 3.2.  The only printable characters are:
680   * <UL>
681   *   <LI>All uppercase and lowercase ASCII alphabetic letters</LI>
682   *   <LI>All ASCII numeric digits</LI>
683   *   <LI>The following additional ASCII characters:  single quote, left
684   *       parenthesis, right parenthesis, plus, comma, hyphen, period, equals,
685   *       forward slash, colon, question mark, space.</LI>
686   * </UL>
687   *
688   * @param  c  The character for which to make the determination.
689   *
690   * @return  {@code true} if the provided character is a printable ASCII
691   *          character, or {@code false} if not.
692   */
693  public static boolean isPrintable(final char c)
694  {
695    if (((c >= 'a') && (c <= 'z')) ||
696        ((c >= 'A') && (c <= 'Z')) ||
697        ((c >= '0') && (c <= '9')))
698    {
699      return true;
700    }
701
702    switch (c)
703    {
704      case '\'':
705      case '(':
706      case ')':
707      case '+':
708      case ',':
709      case '-':
710      case '.':
711      case '=':
712      case '/':
713      case ':':
714      case '?':
715      case ' ':
716        return true;
717      default:
718        return false;
719    }
720  }

769        case ',':
770        case '-':
771        case '.':

929  /**
930   * Indicates whether the contents of the provided array are valid UTF-8.
931   *
932   * @param  b  The byte array to examine.  It must not be {@code null}.
933   *
934   * @return  {@code true} if the byte array can be parsed as a valid UTF-8
935   *          string, or {@code false} if not.
936   */
937  public static boolean isValidUTF8(@NotNull final byte[] b)
938  {
939    int i = 0;
940    while (i < b.length)
941    {
942      final byte currentByte = b[i++];
943
944      // If the most significant bit is not set, then this represents a valid
945      // single-byte character.
946      if ((currentByte & 0b1000_0000) == 0b0000_0000)
947      {
948        continue;
949      }
950
951      // If the first byte starts with 0b110, then it must be followed by
952      // another byte that starts with 0b10.
953      if ((currentByte & 0b1110_0000) == 0b1100_0000)
954      {
955        if (! hasExpectedSubsequentUTF8Bytes(b, i, 1))
956        {
957          return false;
958        }
959
960        i++;
961        continue;
962      }
963
964      // If the first byte starts with 0b1110, then it must be followed by two
965      // more bytes that start with 0b10.
966      if ((currentByte & 0b1111_0000) == 0b1110_0000)
967      {
968        if (! hasExpectedSubsequentUTF8Bytes(b, i, 2))
969        {
970          return false;
971        }
972
973        i += 2;
974        continue;
975      }
976
977      // If the first byte starts with 0b11110, then it must be followed by
978      // three more bytes that start with 0b10.
979      if ((currentByte & 0b1111_1000) == 0b1111_0000)
980      {
981        if (! hasExpectedSubsequentUTF8Bytes(b, i, 3))
982        {
983          return false;
984        }
985
986        i += 3;
987        continue;
988      }
989
990      // If the first byte starts with 0b111110, then it must be followed by
991      // four more bytes that start with 0b10.
992      if ((currentByte & 0b1111_1100) == 0b1111_1000)
993      {
994        if (! hasExpectedSubsequentUTF8Bytes(b, i, 4))
995        {
996          return false;
997        }
998
999        i += 4;
1000        continue;
1001      }
1002
1003      // If the first byte starts with 0b1111110, then it must be followed by
1004      // five more bytes that start with 0b10.
1005      if ((currentByte & 0b1111_1110) == 0b1111_1100)
1006      {
1007        if (! hasExpectedSubsequentUTF8Bytes(b, i, 5))
1008        {
1009          return false;
1010        }
1011
1012        i += 5;
1013        continue;
1014      }
1015
1016      // This is not a valid first byte for a UTF-8 character.
1017      return false;
1018    }
1019
1020
1021    // If we've gotten here, then the provided array represents a valid UTF-8
1022    // string.
1023    return true;
1024  }

1681                                    final int indent,
1682                                    @NotNull final StringBuilder buffer)
1683  {

1685    {
1686      return;
1687    }

1689    for (int i=0; i < indent; i++)
1690    {
1691      buffer.append(' ');
1692    }
1693

1695    int startPos = 0;
1696    while (pos < array.length)
1697    {
1698      toHex(array[pos++], buffer);
1699      buffer.append(' ');
1700
1701      if ((pos % 16) == 0)
1702      {
1703        buffer.append("  ");
1704        for (int i=startPos; i < pos; i++)
1705        {
1706          if ((array[i] < ' ') || (array[i] > '~'))
1707          {
1708            buffer.append(' ');
1709          }
1710          else
1711          {
1712            buffer.append((char) array[i]);
1713          }
1714        }
1715        buffer.append(EOL);
1716        startPos = pos;
1717
1718        if (pos < array.length)
1719        {
1720          for (int i=0; i < indent; i++)
1721          {
1722            buffer.append(' ');
1723          }
1724        }
1725      }
1726    }
1727
1728    // If the last line isn't complete yet, then finish it off.
1729    if ((array.length % 16) != 0)
1730    {
1731      final int missingBytes = (16 - (array.length % 16));
1732      for (int i=0; i < missingBytes; i++)
1733      {
1734        buffer.append("   ");
1735      }
1736      buffer.append("  ");
1737      for (int i=startPos; i < array.length; i++)
1738      {
1739        if ((array[i] < ' ') || (array[i] > '~'))

2370   *
2371   * @return  The {@code Date} object decoded from the provided timestamp.
2372   *
2373   * @throws  ParseException  If the provided string could not be decoded as a
2374   *                          timestamp in generalized time format.
2375   */

2377  public static Date decodeGeneralizedTime(@NotNull final String t)
2378         throws ParseException
2379  {
2380    Validator.ensureNotNull(t);
2381
2382    // Extract the time zone information from the end of the value.
2383    int tzPos;
2384    final TimeZone tz;
2385    if (t.endsWith("Z"))
2386    {
2387      tz = TimeZone.getTimeZone("UTC");
2388      tzPos = t.length() - 1;
2389    }
2390    else
2391    {
2392      tzPos = t.lastIndexOf('-');
2393      if (tzPos < 0)
2394      {
2395        tzPos = t.lastIndexOf('+');
2396        if (tzPos < 0)
2397        {
2398          throw new ParseException(ERR_GENTIME_DECODE_CANNOT_PARSE_TZ.get(t),
2399                                   0);
2400        }
2401      }
2402
2403      tz = TimeZone.getTimeZone("GMT" + t.substring(tzPos));
2404      if (tz.getRawOffset() == 0)
2405      {
2406        // This is the default time zone that will be returned if the value
2407        // cannot be parsed.  If it's valid, then it will end in "+0000" or
2408        // "-0000".  Otherwise, it's invalid and GMT was just a fallback.
2409        if (! (t.endsWith("+0000") || t.endsWith("-0000")))
2410        {
2411          throw new ParseException(ERR_GENTIME_DECODE_CANNOT_PARSE_TZ.get(t),
2412                                   tzPos);
2413        }
2414      }
2415    }
2416
2417
2418    // See if the timestamp has a sub-second portion.  Note that if there is a
2419    // sub-second portion, then we may need to massage the value so that there
2420    // are exactly three sub-second characters so that it can be interpreted as
2421    // milliseconds.
2422    final String subSecFormatStr;
2423    final String trimmedTimestamp;
2424    int periodPos = t.lastIndexOf('.', tzPos);
2425    if (periodPos > 0)
2426    {
2427      final int subSecondLength = tzPos - periodPos - 1;
2428      switch (subSecondLength)
2429      {
2430        case 0:
2431          subSecFormatStr  = "";
2432          trimmedTimestamp = t.substring(0, periodPos);
2433          break;
2434        case 1:
2435          subSecFormatStr  = ".SSS";
2436          trimmedTimestamp = t.substring(0, (periodPos+2)) + "00";
2437          break;
2438        case 2:
2439          subSecFormatStr  = ".SSS";
2440          trimmedTimestamp = t.substring(0, (periodPos+3)) + '0';
2441          break;
2442        default:
2443          subSecFormatStr  = ".SSS";
2444          trimmedTimestamp = t.substring(0, periodPos+4);
2445          break;
2446      }
2447    }
2448    else
2449    {
2450      subSecFormatStr  = "";
2451      periodPos        = tzPos;
2452      trimmedTimestamp = t.substring(0, tzPos);
2453    }
2454
2455
2456    // Look at where the period is (or would be if it existed) to see how many
2457    // characters are in the integer portion.  This will give us what we need
2458    // for the rest of the format string.
2459    final String formatStr;
2460    switch (periodPos)
2461    {
2462      case 10:
2463        formatStr = "yyyyMMddHH" + subSecFormatStr;
2464        break;
2465      case 12:
2466        formatStr = "yyyyMMddHHmm" + subSecFormatStr;
2467        break;
2468      case 14:
2469        formatStr = "yyyyMMddHHmmss" + subSecFormatStr;
2470        break;
2471      default:
2472        throw new ParseException(ERR_GENTIME_CANNOT_PARSE_INVALID_LENGTH.get(t),
2473                                 periodPos);
2474    }
2475
2476
2477    // We should finally be able to create an appropriate date format object
2478    // to parse the trimmed version of the timestamp.
2479    final SimpleDateFormat dateFormat = new SimpleDateFormat(formatStr);
2480    dateFormat.setTimeZone(tz);
2481    dateFormat.setLenient(false);
2482    return dateFormat.parse(trimmedTimestamp);

2532   * Decodes the provided string as a timestamp encoded in the ISO 8601 format
2533   * described in RFC 3339.
2534   *
2535   * @param  timestamp  The timestamp to be decoded in the RFC 3339 format.
2536   *
2537   * @return  The {@code Date} object decoded from the provided timestamp.
2538   *
2539   * @throws  ParseException  If the provided string could not be decoded as a
2540   *                          timestamp in the RFC 3339 time format.
2541   */
2542  @NotNull()
2543  public static Date decodeRFC3339Time(@NotNull final String timestamp)
2544         throws ParseException
2545  {
2546    // Make sure that the string representation has the minimum acceptable
2547    // length.
2548    if (timestamp.length() < 20)
2549    {
2550      throw new ParseException(ERR_RFC_3339_TIME_TOO_SHORT.get(timestamp), 0);
2551    }
2552
2553
2554    // Parse the year, month, day, hour, minute, and second components from the
2555    // timestamp, and make sure the appropriate separator characters are between
2556    // those components.
2557    final int year = parseRFC3339Number(timestamp, 0, 4);
2558    validateRFC3339TimestampSeparatorCharacter(timestamp, 4, '-');
2559    final int month = parseRFC3339Number(timestamp, 5, 2);
2560    validateRFC3339TimestampSeparatorCharacter(timestamp, 7, '-');
2561    final int day = parseRFC3339Number(timestamp, 8, 2);
2562    validateRFC3339TimestampSeparatorCharacter(timestamp, 10, 'T');
2563    final int hour = parseRFC3339Number(timestamp, 11, 2);
2564    validateRFC3339TimestampSeparatorCharacter(timestamp, 13, ':');
2565    final int minute = parseRFC3339Number(timestamp, 14, 2);
2566    validateRFC3339TimestampSeparatorCharacter(timestamp, 16, ':');
2567    final int second = parseRFC3339Number(timestamp, 17, 2);
2568
2569
2570    // Make sure that the month and day values are acceptable.
2571    switch (month)
2572    {
2573      case 1:
2574      case 3:
2575      case 5:
2576      case 7:
2577      case 8:
2578      case 10:
2579      case 12:
2580        // January, March, May, July, August, October, and December all have 31
2581        // days.
2582        if ((day < 1) || (day > 31))
2583        {
2584          throw new ParseException(
2585               ERR_RFC_3339_TIME_INVALID_DAY_FOR_MONTH.get(timestamp, day,
2586                    month),
2587               8);
2588        }
2589        break;
2590
2591      case 4:
2592      case 6:
2593      case 9:
2594      case 11:
2595        // April, June, September, and November all have 30 days.
2596        if ((day < 1) || (day > 30))
2597        {
2598          throw new ParseException(
2599               ERR_RFC_3339_TIME_INVALID_DAY_FOR_MONTH.get(timestamp, day,
2600                    month),
2601               8);
2602        }
2603        break;
2604
2605      case 2:
2606        // February can have 28 or 29 days, depending on whether it's a leap
2607        // year.  Although we could determine whether the provided year is a
2608        // leap year, we'll just always accept up to 29 days for February.
2609        if ((day < 1) || (day > 29))
2610        {
2611          throw new ParseException(
2612               ERR_RFC_3339_TIME_INVALID_DAY_FOR_MONTH.get(timestamp, day,
2613                    month),
2614               8);
2615        }
2616        break;
2617
2618      default:
2619        throw new ParseException(
2620             ERR_RFC_3339_TIME_INVALID_MONTH.get(timestamp, month), 5);
2621    }
2622
2623
2624    // Make sure that the hour, minute, and second values are acceptable.  Note
2625    // that while ISO 8601 permits a value of 24 for the hour, RFC 3339 only
2626    // permits hour values between 0 and 23.  Also note that some minutes can
2627    // have up to 61 seconds for leap seconds, so we'll always account for that.
2628    if ((hour < 0) || (hour > 23))
2629    {
2630      throw new ParseException(
2631           ERR_RFC_3339_TIME_INVALID_HOUR.get(timestamp, hour), 11);
2632    }
2633
2634    if ((minute < 0) || (minute > 59))
2635    {
2636      throw new ParseException(
2637           ERR_RFC_3339_TIME_INVALID_MINUTE.get(timestamp, minute), 14);
2638    }
2639
2640    if ((second < 0) || (second > 60))
2641    {
2642      throw new ParseException(
2643           ERR_RFC_3339_TIME_INVALID_SECOND.get(timestamp, second), 17);
2644    }
2645
2646
2647    // See if there is a sub-second portion.  If so, then there will be a
2648    // period at position 19 followed by at least one digit.  This
2649    // implementation will only support timestamps with no more than three
2650    // sub-second digits.
2651    int milliseconds = 0;
2652    int timeZoneStartPos = -1;
2653    if (timestamp.charAt(19) == '.')
2654    {
2655      int numDigits = 0;
2656      final StringBuilder subSecondString = new StringBuilder(3);
2657      for (int pos=20; pos < timestamp.length(); pos++)
2658      {
2659        final char c = timestamp.charAt(pos);
2660        switch (c)
2661        {
2662          case '0':
2663            numDigits++;
2664            if (subSecondString.length() > 0)
2665            {
2666              // Only add a zero if it's not the first digit.
2667              subSecondString.append(c);
2668            }
2669            break;
2670          case '1':
2671          case '2':
2672          case '3':
2673          case '4':
2674          case '5':
2675          case '6':
2676          case '7':
2677          case '8':
2678          case '9':
2679            numDigits++;
2680            subSecondString.append(c);
2681            break;
2682          case 'Z':
2683          case '+':
2684          case '-':
2685            timeZoneStartPos = pos;
2686            break;
2687          default:
2688            throw new ParseException(
2689                 ERR_RFC_3339_TIME_INVALID_SUB_SECOND_CHAR.get(timestamp, c,
2690                      pos),
2691                 pos);
2692        }
2693
2694        if (timeZoneStartPos > 0)
2695        {
2696          break;
2697        }
2698
2699        if (numDigits > 3)
2700        {
2701          throw new ParseException(
2702               ERR_RFC_3339_TIME_TOO_MANY_SUB_SECOND_DIGITS.get(timestamp),
2703               20);
2704        }
2705      }
2706
2707      if (timeZoneStartPos < 0)
2708      {
2709        throw new ParseException(
2710             ERR_RFC_3339_TIME_MISSING_TIME_ZONE_AFTER_SUB_SECOND.get(
2711                  timestamp),
2712             (timestamp.length() - 1));
2713      }
2714
2715      if (numDigits == 0)
2716      {
2717        throw new ParseException(
2718             ERR_RFC_3339_TIME_NO_SUB_SECOND_DIGITS.get(timestamp), 19);
2719      }
2720
2721      if (subSecondString.length() == 0)
2722      {
2723        // This is possible if the sub-second portion is all zeroes.
2724        subSecondString.append('0');
2725      }
2726
2727      milliseconds = Integer.parseInt(subSecondString.toString());
2728      if (numDigits == 1)
2729      {
2730        milliseconds *= 100;
2731      }
2732      else if (numDigits == 2)
2733      {
2734        milliseconds *= 10;
2735      }
2736    }
2737    else
2738    {
2739      timeZoneStartPos = 19;
2740    }
2741
2742
2743    // The remainder of the timestamp should be the time zone.
2744    final TimeZone timeZone;
2745    if (timestamp.substring(timeZoneStartPos).equals("Z"))
2746    {
2747      // This is shorthand for the UTC time zone.
2748      timeZone = UTC_TIME_ZONE;
2749    }
2750    else
2751    {
2752      // This is an offset from UTC, which should be in the form "+HH:MM" or
2753      // "-HH:MM".  Make sure it has the expected length.
2754      if ((timestamp.length() - timeZoneStartPos) != 6)
2755      {
2756        throw new ParseException(
2757             ERR_RFC_3339_TIME_INVALID_TZ.get(timestamp), timeZoneStartPos);
2758      }
2759
2760      // Make sure it starts with "+" or "-".
2761      final int firstChar = timestamp.charAt(timeZoneStartPos);
2762      if ((firstChar != '+') && (firstChar != '-'))
2763      {
2764        throw new ParseException(
2765             ERR_RFC_3339_TIME_INVALID_TZ.get(timestamp), timeZoneStartPos);
2766      }
2767
2768
2769      // Make sure the hour offset is valid.
2770      final int timeZoneHourOffset =
2771           parseRFC3339Number(timestamp, (timeZoneStartPos+1), 2);
2772      if ((timeZoneHourOffset < 0) || (timeZoneHourOffset > 23))
2773      {
2774        throw new ParseException(
2775             ERR_RFC_3339_TIME_INVALID_TZ.get(timestamp), timeZoneStartPos);
2776      }
2777
2778
2779      // Make sure there is a colon between the hour and the minute portions of
2780      // the offset.
2781      if (timestamp.charAt(timeZoneStartPos+3) != ':')
2782      {
2783        throw new ParseException(
2784             ERR_RFC_3339_TIME_INVALID_TZ.get(timestamp), timeZoneStartPos);
2785      }
2786
2787      final int timeZoneMinuteOffset =
2788           parseRFC3339Number(timestamp, (timeZoneStartPos+4), 2);
2789      if ((timeZoneMinuteOffset < 0) || (timeZoneMinuteOffset > 59))
2790      {
2791        throw new ParseException(
2792             ERR_RFC_3339_TIME_INVALID_TZ.get(timestamp), timeZoneStartPos);
2793      }
2794
2795      timeZone = TimeZone.getTimeZone(
2796           "GMT" + timestamp.substring(timeZoneStartPos));
2797    }
2798
2799
2800    // Put everything together to construct the appropriate date.
2801    final GregorianCalendar calendar =
2802         new GregorianCalendar(year,
2803              (month-1), // NOTE:  Calendar stupidly uses zero-indexed months.
2804              day, hour, minute, second);
2805    calendar.set(GregorianCalendar.MILLISECOND, milliseconds);
2806    calendar.setTimeZone(timeZone);
2807    return calendar.getTime();
2808  }
