061/**
062 * This class provides an {@link LDAPResultWriter} instance that simply prints
063 * the values of all attributes in the entries that are returned, with each
064 * value on a line by itself.  Entry DNs will not be included, and there will
065 * not be any blank lines between entries.
066 * <BR>
067 * <BLOCKQUOTE>
068 *   <B>NOTE:</B>  This class, and other classes within the
069 *   {@code com.unboundid.ldap.sdk.unboundidds} package structure, are only
070 *   supported for use against Ping Identity, UnboundID, and
071 *   Nokia/Alcatel-Lucent 8661 server products.  These classes provide support
072 *   for proprietary functionality or for external specifications that are not
073 *   considered stable or mature enough to be guaranteed to work in an
074 *   interoperable way with other types of LDAP servers.
075 * </BLOCKQUOTE>
076 */
077@ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
078public final class ValuesOnlyLDAPResultWriter
079       extends LDAPResultWriter
080{
081  /**
082   * Creates a new instance of this LDAP result writer.
083   *
084   * @param  outputStream  The output stream to which the output will be
085   *                       written.
086   */
087  public ValuesOnlyLDAPResultWriter(@NotNull final OutputStream outputStream)
088  {
089    super(outputStream);
090  }
091
092
093
094  /**
095   * {@inheritDoc}
096   */
097  @Override()
098  public void writeComment(@NotNull final String comment)
099  {
100    // Comments will not be written in this format.
101  }
102
103
104
105  /**
106   * {@inheritDoc}
107   */
108  @Override()
109  public void writeHeader()
110  {
111    // No header is required for this format.
112  }
113
114
115
116  /**
117   * {@inheritDoc}
118   */
119  @Override()
120  public void writeSearchResultEntry(@NotNull final SearchResultEntry entry)
121  {
122    try
123    {
124      for (final Attribute a : entry.getAttributes())
125      {
126        for (final byte[] value : a.getValueByteArrays())
127        {
128          getPrintStream().write(value);
129          getPrintStream().write(StaticUtils.EOL_BYTES);
130        }
131      }
132    }
133    catch (final Exception e)
134    {
135      Debug.debugException(e);
136      throw new LDAPRuntimeException(new LDAPException(ResultCode.OTHER,
137           ERR_VALUES_ONLY_OUTPUT_FORMAT_WRITE_ERROR.get(
138                StaticUtils.getExceptionMessage(e)),
139           e));
140    }
141  }
142
143
144
145  /**
146   * {@inheritDoc}
147   */
148  @Override()
149  public void writeSearchResultReference(
150                   @NotNull final SearchResultReference ref)
151  {
152    // No output is needed for search result reference messages.
153  }
154
155
156
157  /**
158   * {@inheritDoc}
159   */
160  @Override()
161  public void writeResult(@NotNull final LDAPResult result)
162  {
163    getPrintStream().flush();
164  }
165
166
167
168  /**
169   * {@inheritDoc}
170   */
171  @Override()
172  public void writeUnsolicitedNotification(
173                   @NotNull final LDAPConnection connection,
174                   @NotNull final ExtendedResult notification)
175  {
176    getPrintStream().flush();
177  }
178}
