377  private final int intValue;

380  // usability error.
381  @Nullable private final String message;
382

384  @NotNull private final String name;
385
386  // The encoded string representation for this account usability error.
387  @NotNull private final String stringRepresentation;
388
389
390
391  /**
392   * Creates a new authentication failure reason with the provided information.
393   *
394   * @param  intValue  The integer value for this authentication failure reason.
395   * @param  name      The name for this authentication failure reason.  It must
396   *                   not be {@code null}.
397   * @param  message   A human-readable message that provides specific details
398   *                   about this account usability error.  It may be
399   *                   {@code null} if no message is available.
400   */
401  public AuthenticationFailureReason(final int intValue,
402                                     @NotNull final String name,
403                                     @Nullable final String message)
404  {
405    Validator.ensureNotNull(name);
406
407    this.intValue = intValue;
408    this.name = name;
409    this.message = message;
410
411    final StringBuilder buffer = new StringBuilder();
412    buffer.append("code=");
413    buffer.append(intValue);
414    buffer.append("\tname=");
415    buffer.append(name);
416
417    if (message != null)
418    {
419      buffer.append("\tmessage=");
420      buffer.append(message);
421    }
422
423    stringRepresentation = buffer.toString();
424  }
425
426
427
428  /**
429   * Creates a new authentication failure reason that is decoded from the
430   * provided string representation.
431   *
432   * @param  stringRepresentation  The string representation of the
433   *                               authentication failure reason to decode.  It
434   *                               must not be {@code null}.
435   *
436   * @throws LDAPException  If the provided string cannot be decoded as a valid
437   *                         authentication failure reason.
438   */
439  public AuthenticationFailureReason(@NotNull final String stringRepresentation)
440       throws LDAPException
441  {
442    this.stringRepresentation = stringRepresentation;
443
444    try
445    {
446      Integer i = null;
447      String n = null;
448      String m = null;
449
450      final StringTokenizer tokenizer =
451           new StringTokenizer(stringRepresentation, "\t");
452      while (tokenizer.hasMoreTokens())
453      {
454        final String token = tokenizer.nextToken();
455        final int equalPos = token.indexOf('=');
456        final String fieldName = token.substring(0, equalPos);
457        final String fieldValue = token.substring(equalPos+1);
458        if (fieldName.equals("code"))
459        {
460          i = Integer.valueOf(fieldValue);
461        }
462        else if (fieldName.equals("name"))
463        {
464          n = fieldValue;
465        }
466        else if (fieldName.equals("message"))
467        {
468          m = fieldValue;
469        }
470      }
471
472      if (i == null)
473      {
474        throw new LDAPException(ResultCode.DECODING_ERROR,
475             ERR_AUTH_FAILURE_REASON_CANNOT_DECODE.get(stringRepresentation,
476                  ERR_AUTH_FAILURE_REASON_NO_CODE.get()));
477      }
478
479      if (n == null)
480      {
481        throw new LDAPException(ResultCode.DECODING_ERROR,
482             ERR_AUTH_FAILURE_REASON_CANNOT_DECODE.get(stringRepresentation,
483                  ERR_AUTH_FAILURE_REASON_NO_NAME.get()));
484      }
485
486      intValue = i;
487      name     = n;
488      message  = m;
489    }
490    catch (final LDAPException le)
491    {
492      Debug.debugException(le);
493
494      throw le;
495    }
496    catch (final Exception e)
497    {
498      Debug.debugException(e);
499
500      throw new LDAPException(ResultCode.DECODING_ERROR,
501           ERR_AUTH_FAILURE_REASON_CANNOT_DECODE.get(stringRepresentation,
502                StaticUtils.getExceptionMessage(e)),
503           e);
504    }
505  }
